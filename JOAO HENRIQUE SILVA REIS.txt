<?php
    class Product {
        private $id;
        private $name;
        private $price;

        // Construtor
        public function __construct($id, $name, $price) {
            $this->id = $id;
            $this->name = $name;
            $this->price = $price;
        }

        // MÃ©todos getters e setters
        public function getId() {
            return $this->id;
        }

        public function getName() {
            return $this->name;
        }

        public function setName($name) {
            $this->name = $name;
        }

        public function getPrice() {
            return $this->price;
        }

        public function setPrice($price) {
            $this->price = $price;
        }
    }
?>
